ICliPanelRef 
export interface ICliPanelRef<TEngine = unknown> {
    // Panel chrome
    open(): void;
    collapse(): void;
    toggleCollapse(): void;
    hide(): void;
    unhide(): void;
    toggleHide(): void;
    close(): void;
    maximize(): void;
    restore(): void;
    toggleMaximize(): void;
    resize(dimensions: { height?: number; width?: number }): void;
    setPosition(position: CliPanelPosition): void;

    // Tabs
    addTab(title?: string): number;
    closeTab(tabId: number): void;
    selectTab(tabId: number): void;
    renameTab(tabId: number, title: string): void;

    // Panes
    splitPane(tabId?: number): number;
    closePane(paneId: number): void;

    // Engine access
    getEngine(paneId?: number): TEngine | undefined;

    // State
    getState(): CliPanelState;
}

import { Component, ViewChild } from '@angular/core';
import { CliPanelComponent } from '@qodalis/angular-cli';

@Component({
    selector: 'app-root',
    templateUrl: './app.component.html',
})
export class AppComponent {
    @ViewChild('panel') panel!: CliPanelComponent;

    openPanel() {
        this.panel.open();
    }

    addLogsTab() {
        const tabId = this.panel.addTab('Logs');
        this.panel.selectTab(tabId);
    }

    inspectState() {
        const state = this.panel.getState();
        console.log('Panel state:', state);
    }
}

App.tsx 
import { useRef } from 'react';
import { CliPanel } from '@qodalis/react-cli';
import type { ICliPanelRef } from '@qodalis/cli-core';
import type { CliEngine } from '@qodalis/cli';

function App() {
    const panelRef = useRef<ICliPanelRef<CliEngine>>(null);

    return (
        <>
            <button onClick={() => panelRef.current?.open()}>
                Open Panel
            </button>
            <button onClick={() => panelRef.current?.addTab('Logs')}>
                Add Logs Tab
            </button>

            <CliPanel
                ref={panelRef}
                options={options}
                modules={modules}
            />
        </>
    );
}

App.tsx 
import { useState } from 'react';
import { CliPanel } from '@qodalis/react-cli';

function App() {
    const [collapsed, setCollapsed] = useState(true);
    const [position, setPosition] = useState('bottom');

    return (
        <>
            <button onClick={() => setCollapsed(c => !c)}>
                Toggle Panel
            </button>

            <CliPanel
                collapsed={collapsed}
                onCollapsedChange={setCollapsed}
                position={position}
                onPositionChange={setPosition}
                onTabAdded={({ tabId }) => console.log('New tab:', tabId)}
                options={options}
                modules={modules}
            />
        </>
    );
}
