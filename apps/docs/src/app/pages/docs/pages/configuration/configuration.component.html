App.tsx 
import { Cli, CliPanel, CliConfigProvider } from '@qodalis/react-cli';

function App() {
    const options = {
        logLevel: 'DEBUG',
        welcomeMessage: 'Welcome to my app!',
        theme: 'default',
        serverUrl: 'http://localhost:8046',
    };

    return (
        <CliConfigProvider modules={[]} options={options}>
            <Cli />
        </CliConfigProvider>
    );
}
