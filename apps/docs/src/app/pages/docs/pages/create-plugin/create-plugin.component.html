import {
    ICliCommandProcessor,
    ICliExecutionContext,
    CliProcessCommand,
} from '@qodalis/cli-core';

export class WeatherProcessor implements ICliCommandProcessor {
    command = 'weather';
    description = 'Get weather forecasts';

    parameters = [
        {
            name: 'city',
            description: 'City name to look up',
            required: true,
        },
        {
            name: '--units',
            description: 'Temperature units (metric or imperial)',
            required: false,
        },
    ];

    async processCommand(
        command: CliProcessCommand,
        context: ICliExecutionContext,
    ): Promise<void> {
        const city = command.args[0];
        if (!city) {
            context.writer.writeError('Please provide a city name.');
            return;
        }

        context.spinner.show(`Fetching weather for ${city}...`);
        // Your logic here
        context.spinner.hide();
        context.writer.writeln(`Weather in ${city}: Sunny, 22C`);
    }
}
