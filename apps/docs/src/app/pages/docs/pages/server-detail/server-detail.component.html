import { CliJobsBuilder } from '@qodalis/cli-server-plugin-jobs';

const jobsPlugin = new CliJobsBuilder()
    .addJob(new HealthCheckJob(), {
        name: 'health-check',
        description: 'Periodic health check',
        interval: '30s',
    })
    .build((msg) => eventSocketManager.broadcastMessage(msg));

app.use('/api/v1/qcli/jobs', jobsPlugin.router);
await jobsPlugin.scheduler.start();

server.py 
from qodalis_cli_jobs import CliJobsBuilder
from qodalis_cli_server_abstractions.jobs import CliJobOptions

jobs_plugin = (
    CliJobsBuilder()
    .add_job(
        HealthCheckJob(),
        CliJobOptions(
            name="health-check",
            description="Periodic health check",
            interval="30s",
        ),
    )
    .build(broadcast_fn=event_socket_manager.broadcast_message)
)

app.include_router(jobs_plugin.router, prefix="/api/v1/qcli/jobs")
await jobs_plugin.scheduler.start()
