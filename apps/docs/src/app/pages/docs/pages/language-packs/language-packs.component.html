lang-pack-example.ts 
export class SpanishLanguageProcessor implements ICliCommandProcessor {
    command = '__lang_es__';
    description = 'Spanish language pack';

    onInit(context: ICliExecutionContext): void {
        const translator = context.translationService;
        translator.addTranslations('es', {
            'cli.help.description': 'Muestra la ayuda',
            'cli.clear.description': 'Limpia la terminal',
            'cli.version.description': 'Muestra la version',
            // ... more translations
        });
    }
}
