app.component.ts 
import { Component, ViewChild } from '@angular/core';
import { CliComponent } from '@qodalis/angular-cli';
import { CliEngine } from '@qodalis/cli';

@Component({
    selector: 'app-root',
    templateUrl: './app.component.html',
})
export class AppComponent {
    private engine?: CliEngine;

    // Option A: capture via event
    onEngineReady(engine: CliEngine) {
        this.engine = engine;
    }

    // Option B: access via ViewChild
    @ViewChild('terminal') terminal!: CliComponent;

    async runCommand(command: string) {
        // Use the captured engine
        await this.engine?.execute(command);

        // Or get it from the component
        const engine = this.terminal.getEngine();
        await engine?.execute(command);
    }
}

App.tsx 
import { useRef } from 'react';
import { Cli, CliPanel } from '@qodalis/react-cli';
import type { ICliPanelRef } from '@qodalis/cli-core';
import type { CliEngine } from '@qodalis/cli';

function App() {
    const engineRef = useRef<CliEngine | null>(null);
    const panelRef = useRef<ICliPanelRef<CliEngine>>(null);

    async function runCommand(command: string) {
        // From standalone <Cli> via onEngineReady
        await engineRef.current?.execute(command);
    }

    async function runOnPanel(command: string) {
        // From <CliPanel> via getEngine()
        panelRef.current?.open();
        const engine = panelRef.current?.getEngine();
        await engine?.execute(command);
    }

    return (
        <>
            <button onClick={() => runCommand('help')}>Help</button>

            <Cli
                options={options}
                modules={modules}
                onEngineReady={(e) => (engineRef.current = e)}
            />

            <CliPanel ref={panelRef} options={panelOptions} />
        </>
    );
}

main.ts 
import '@qodalis/cli/assets/styles.css';
import { CliEngine } from '@qodalis/cli';
import { guidModule } from '@qodalis/cli-guid';

const container = document.getElementById('terminal')!;

const engine = new CliEngine(container, {
    welcomeMessage: 'Ready!',
});

engine.registerModules([guidModule]);

await engine.start();
engine.focus();

// Execute commands directly
await engine.execute('help');
await engine.execute('guid');
