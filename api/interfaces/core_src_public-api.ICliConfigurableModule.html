Interface ICliConfigurableModule<T>

A module that accepts typed configuration via configure(). Extend this instead of ICliModule when your module has a config type.

Example

interface MyModuleConfig { verbose?: boolean }
const myModule: ICliConfigurableModule<MyModuleConfig> = {
    configure(config) { return { ...this, config }; },
    // ...
};
interface ICliConfigurableModule<
    T extends Record<string, any> = Record<string, any>,
> {
    apiVersion: number;
    config?: Record<string, any>;
    dependencies?: string[];
    description?: string;
    name: string;
    priority?: number;
    processors?: ICliCommandProcessor[];
    services?: CliProvider[];
    translations?: Record<string, Record<string, string>>;
    version?: string;
    configure(config: T): ICliModule;
    onAfterBoot(context: ICliExecutionContext): Promise<void>;
    onDestroy(context: ICliExecutionContext): Promise<void>;
    onInit(context: ICliExecutionContext): Promise<void>;
    onSetup(context: ICliExecutionContext): Promise<boolean>;
}

Properties

apiVersion

apiVersion: number

The web plugin API version this module targets.

This is a frontend-only concept used by the browser CLI runtime to ensure plugin compatibility. It has no relation to backend server API versioning (e.g. /api/v1/qcli/).

Modules with apiVersion less than the runtime's API_VERSION are rejected with a warning at boot time.

Use the API_VERSION constant exported from @qodalis/cli-core to target the current runtime version.

Optionaltranslations

translations?: Record<string, Record<string, string>>

Per-locale translation maps shipped by this module. Keyed by locale code (e.g. 'es', 'fr'), each value is a flat key-value map of translation keys to translated strings.

These are automatically registered with the translation service during module boot — no manual onInit() code required.

Example

translations: {
  es: { 'cli.guid.description': 'Generar y validar UUIDs' },
  fr: { 'cli.guid.description': 'Générer et valider des UUIDs' },
}
