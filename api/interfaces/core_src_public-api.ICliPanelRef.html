Interface ICliPanelRef<TEngine>

Framework-agnostic interface for programmatic control of the CLI panel.

CliEngine is referenced as a generic to avoid a circular dependency between @qodalis/cli-core and @qodalis/cli. Framework implementations resolve this to the concrete CliEngine class.

interface ICliPanelRef<TEngine = unknown> {
    addTab(title?: string): number;
    close(): void;
    closePane(paneId: number): void;
    closeTab(tabId: number): void;
    collapse(): void;
    getEngine(paneId?: number): undefined | TEngine;
    getState(): CliPanelState;
    hide(): void;
    maximize(): void;
    open(): void;
    renameTab(tabId: number, title: string): void;
    resize(dimensions: { height?: number; width?: number }): void;
    restore(): void;
    selectTab(tabId: number): void;
    setPosition(position: CliPanelPosition): void;
    splitPane(tabId?: number): number;
    toggleCollapse(): void;
    toggleHide(): void;
    toggleMaximize(): void;
    unhide(): void;
}
