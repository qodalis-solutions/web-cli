Interface ICliFileTransferService

Abstraction for file transfer operations. Default implementation uses browser-native APIs (file picker, download dialog). @qodalis/cli-files overrides with virtual FS backed by IndexedDB.

interface ICliFileTransferService {
    downloadToBrowser(filename: string, content: string | Blob): void;
    exists(path: string): Promise<boolean>;
    listFiles(path: string): Promise<ICliFileEntry[]>;
    readFile(path: string): Promise<null | string>;
    uploadFromBrowser(
        accept?: string,
    ): Promise<null | { content: string; name: string }>;
    writeFile(path: string, content: string): Promise<void>;
}
