Interface ICliInputReader

Provides interactive input reading from the terminal. All methods resolve with null when the user aborts (Ctrl+C or Escape).

interface ICliInputReader {
    readConfirm(
        prompt: string,
        defaultValue?: boolean,
    ): Promise<null | boolean>;
    readDate(prompt: string, options?: CliDateOptions): Promise<null | string>;
    readFile(
        prompt: string,
        options?: CliFilePickerOptions,
    ): Promise<null | CliFileResult[]>;
    readLine(prompt: string, options?: CliLineOptions): Promise<null | string>;
    readMultiSelect(
        prompt: string,
        options: CliMultiSelectOption[],
        selectOptions?: CliMultiSelectOptions,
    ): Promise<null | string[]>;
    readNumber(
        prompt: string,
        options?: { default?: number; max?: number; min?: number },
    ): Promise<null | number>;
    readPassword(prompt: string): Promise<null | string>;
    readSelect(
        prompt: string,
        options: CliSelectOption[],
        selectOptions?: CliSelectOptions,
    ): Promise<null | string>;
    readSelectInline(
        prompt: string,
        options: CliSelectOption[],
        selectOptions?: CliSelectOptions,
    ): Promise<null | string>;
}
